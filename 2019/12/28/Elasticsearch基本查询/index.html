查询

基本查询

  1. match(会对输入进行分词)
GET lagou/_search
{
  "query": {
    "match": {
      "title": "爬取"
    }
  }
}

GET lagou/_search
{
  "query": {
    "match": {
      "title": "爬取"
    }
  }
}
  1. term(不会分词)
GET lagou/_search
{
  "query": {
    "term": {
      "title": "python爬虫"
    }
  }
}
  1. terms(满足任何一个)
# terms查询
GET lagou/_search
{
  "query": {
    "terms": {
      "title": ["工程师", "django", "系统"]
    }
  }
}
  1. query查询(控制查询的返回数量)
GET lagou/_search
{
  "query": {
    "match": {
      "title": "python"
    }
  },
  "from":1,
  "size":2
}
  1. match_all查询
GET lagou/_search
{
  "query": {
    "match_all": {}
  }
}
  1. multi_match查询
# 比如可以指定多个字段
# 比如查询title和desc这两个字段里面包含python的关键词文档
# 可以设置权重
GET lagou/_search
{
  "query": {
    "multi_match": {
      "query": "python",
      "fields": ["title^3", "desc"]
    }
  }
}
  1. stored_fields指定返回的字段(mappings设置了store的)
GET lagou/_search
{
  "stored_fields": ["title"],
  "query": {
    "match": {
      "title": "python"
    }
  }
}
  1. sort(排序返回 asc,desc)
GET lagou/_search
{
  "query": {
    "match_all": {}
  },
  "sort": [
    {
      "comments": {
        "order": "desc"
      }
    }
  ]
}
  1. range(范围查询)
    boost: 权重
GET lagou/_search
{
  "query": {
    "range": {
      "comments": {
        "gte": 10,
        "lte": 20,
        "boost": 2.0
      }
    }
  }
}

GET lagou/_search
{
  "query": {
    "range": {
      "add_time": {
        "gt": "2019-4-1"
      }
    }
  }
}
  1. match_phrase(短语查询，自动分词，满足所有则返回)
    slop: 两个词之前的距离
GET lagou/_search
{
  "query": {
    "match_phrase": {
      "title": {
        "query": "python系统",
        "slop": 6
      }
    }
  }
}
